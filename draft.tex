    {\printtext[parens]{\printlabeldateextra}},  % comma is printed
  \printtext[bibhyperlink]{%  make it a link to citation
    \bibstring{backrefpage}\ppspace%  "cyt. na s. "
    \printlist[][-1]{pageref}}%  page number of first occurrence
